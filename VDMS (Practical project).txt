import mysql.connector

def Add_Record():
    Payment=0
    L=[]
    park_no=int(input("Enter the parking number:"))
    L.append(park_no)
    park_name=input("Enter the Parking Name:")
    L.append(park_name)
    block=input("Enter the parking level/block:")
    L.append(block)
    freespace=input("Is there any freespace? (yes/no):")
    L.append(freespace)
    v_no=int(input("Enter the vehicle number:"))
    L.append(v_no)
    Days=int(input("Enter total days of parking:"))
    L.append(Days)
    if Days==1:
        Payment=10
    elif Days==2:
        Payment=30
    elif Days==3:
        Payment=50
    elif Days==4:
        Payment=70
    elif Days==5:
        Payment=90
    elif Days==6:
        Payment=140
    L.append(Payment)
    stud=(L)
    sql='insert into park_details(park_no,park_name,block,freespace,v_no,Days,Payment)values(%s,%s,%s,%s,%s,%s,%s)'
    mycursor.execute(sql,stud)
    mydb.commit()
    print("Parking details are added successfully\n")
